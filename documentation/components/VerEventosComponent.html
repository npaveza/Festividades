import { CommonModule } from '@angular/common';
import { Component, OnInit } from '@angular/core';
import { FormsModule } from '@angular/forms';

interface Evento {
  nombre: string;
  lugar: string;
  fecha: string;
  hora: string;
  puestos?: string;
  entrada?: number;
  estacionamiento?: boolean;
  comentarios?: string[];
}


@Component({
  selector: 'app-ver-eventos',
  standalone: true,
  imports: [CommonModule, FormsModule],
  templateUrl: './ver-eventos.component.html',
  styleUrl: './ver-eventos.component.css'
})

export class VerEventosComponent implements OnInit {
  eventos: Evento[] = [];
  eventoSeleccionado: Evento | null = null;
  nuevoComentario: string = '';

  ngOnInit(): void {
    // Cargar eventos desde LocalStorage al iniciar el componente
    const eventosGuardados = JSON.parse(localStorage.getItem('eventos') || '[]');
    this.eventos = eventosGuardados;
  }

  abrirModal(evento: Evento): void {
    // Asigna el evento seleccionado para mostrar sus detalles en el modal
    this.eventoSeleccionado = evento;
  }

  agregarComentario(): void {
    if (this.eventoSeleccionado && this.nuevoComentario) {
      // Agregar el nuevo comentario al evento seleccionado
      if (!this.eventoSeleccionado.comentarios) {
        this.eventoSeleccionado.comentarios = [];
      }
      this.eventoSeleccionado.comentarios.push(this.nuevoComentario);
      this.nuevoComentario = '';

      // Guardar los cambios en LocalStorage
      localStorage.setItem('eventos', JSON.stringify(this.eventos));
    }
  }
}
